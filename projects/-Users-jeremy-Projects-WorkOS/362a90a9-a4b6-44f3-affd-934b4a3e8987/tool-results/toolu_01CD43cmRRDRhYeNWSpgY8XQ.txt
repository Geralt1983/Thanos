    14→      .select()

    18→      return NextResponse.json({ error: "Move not found" }, { status: 404 })

    21→    const currentIndex = statusOrder.indexOf(move.status)
    22→    if (currentIndex <= 0) {
    23→      return NextResponse.json(move)
    24→    }
    25→
    26→    const newStatus = statusOrder[currentIndex - 1]
    27→    const [updated] = await db
    28→      .update(moves)
    29→      .set({
    30→        status: newStatus,
    31→        updatedAt: new Date(),
    32→      })
    33→      .where(eq(moves.id, Number.parseInt(id, 10)))
    34→      .returning()
    35→
    36→    return NextResponse.json(updated)
    37→  } catch (error) {
    38→    console.error("Failed to demote move:", error)

    40→  }
    41→}
